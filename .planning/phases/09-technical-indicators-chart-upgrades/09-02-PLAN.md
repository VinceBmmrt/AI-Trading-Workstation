       useEffect(() =&gt; {
         if (!showRSI || !rsiContainerRef.current) return;
         let cleanup: (() =&gt; void) | undefined;
         import("lightweight-charts").then(({ createChart, LineSeries }) =&gt; {
           if (!rsiContainerRef.current) return;
           const rsiChart: any = createChart(rsiContainerRef.current, {
             layout: { background: { color: "#161b22" }, textColor: "#7d8590",
                        fontFamily: "'IBM Plex Mono', monospace", fontSize: 10 },
             grid: { vertLines: { color: "#1c2128" }, horzLines: { color: "#1c2128" } },
             rightPriceScale: { borderColor: "#21262d", textColor: "#7d8590",
                                 scaleMargins: { top: 0.1, bottom: 0.1 } },
             timeScale: { borderColor: "#21262d", timeVisible: false, visible: false },
             crosshair: { horzLine: { visible: true }, vertLine: { visible: false } },
             handleScale: false, handleScroll: false,
           });
           const rsiSer: any = rsiChart.addSeries(LineSeries, {
             color: "#209dd7",
             lineWidth: 1,
             priceLineVisible: false,
             lastValueVisible: true,
             crosshairMarkerVisible: true,
             crosshairMarkerRadius: 3,
           });
           rsiSer.createPriceLine({ price: 70, color: "#f85149", lineStyle: 2, lineWidth: 1, axisLabelVisible: true });
           rsiSer.createPriceLine({ price: 50, color: "#7d8590", lineStyle: 2, lineWidth: 1, axisLabelVisible: false });
           rsiSer.createPriceLine({ price: 30, color: "#3fb950", lineStyle: 2, lineWidth: 1, axisLabelVisible: true });
           rsiChart.priceScale("right").applyOptions({ autoScale: false, minimum: 0, maximum: 100 });

           const hist = history.get(ticker) ?? [];
           const rsiVals = calcRSI(hist, 14);
           const now = Math.floor(Date.now() / 1000);
           const seen = new Set&lt;number&gt;();
           const rsiData = rsiVals
             .map((v, i) =&gt; ({ time: now - (hist.length - 1 - i), value: v }))
             .filter((d): d is { time: number; value: number } =&gt; d.value !== null)
             .filter((d) =&gt; { if (seen.has(d.time)) return false; seen.add(d.time); return true; });
           rsiSer.setData(rsiData);
           rsiSerRef.current  = rsiSer;
           rsiChartRef.current = rsiChart;

           const ro = new ResizeObserver(() =&gt; {
             if (rsiContainerRef.current) {
               rsiChart.applyOptions({
                 width: rsiContainerRef.current.clientWidth,
                 height: rsiContainerRef.current.clientHeight,
               });
             }
           });
           ro.observe(rsiContainerRef.current);
           cleanup = () =&gt; {
             ro.disconnect();
             rsiChart.remove();
             rsiSerRef.current  = null;
             rsiChartRef.current = null;
           };
         });
         return () =&gt; cleanup?.();
       }, [ticker, showRSI, history]);

    4. In the live update useEffect (depends on [ticker, prices]), after volume update, add RSI update:
       if (rsiSerRef.current) {

         if (lastRSI !== null) rsiSerRef.current.update({ time, value: lastRSI });
       }
  </action>
