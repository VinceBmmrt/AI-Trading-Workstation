    import { useCallback, useEffect, useState } from "react";
    import { fetchPortfolio, fetchHistory, fetchTrades, fetchAnalytics } from "@/lib/api";
    import type { Portfolio, HistoryPoint, TradeRecord, PortfolioAnalytics } from "@/lib/types";

    export function usePortfolio() {
      const [portfolio, setPortfolio] = useState<Portfolio | null>(null);
      const [history, setHistory]     = useState<HistoryPoint[]>([]);
      const [trades, setTrades]       = useState<TradeRecord[]>([]);
      const [analytics, setAnalytics] = useState<PortfolioAnalytics | null>(null);

      const refresh = useCallback(async () => {
        try {
          const [p, h, t, a] = await Promise.all([
            fetchPortfolio(),
            fetchHistory(),
            fetchTrades(),
            fetchAnalytics(),
          ]);
          setPortfolio(p);
          setHistory(h);
          setTrades(t);
          setAnalytics(a);
        } catch (e) {
          console.error("Portfolio fetch failed", e);
        }
      }, []);

      useEffect(() => {
        refresh();
        const id = setInterval(refresh, 10_000);
        return () => clearInterval(id);
      }, [refresh]);

      return { portfolio, history, trades, analytics, refresh };
    }

    All four fetches are bundled into a single Promise.all. The same refresh callback is called by page.tsx after every trade — trades and analytics auto-refresh after each trade execution without any extra wiring.

    import type { TradeRecord } from "@/lib/types";

    const LABEL = "text-[9px] font-mono text-text-dim uppercase tracking-widest";
    const CELL  = "font-mono text-xs tabular-nums";

    function exportToCsv(trades: TradeRecord[]) {
      if (typeof window === "undefined") return;
      const header = "Date,Ticker,Side,Quantity,Price,Total\n";
      const rows = trades
        .map((t) => `${t.executed_at},${t.ticker},${t.side},${t.quantity},${t.price},${t.total}`)
        .join("\n");
      const blob = new Blob([header + rows], { type: "text/csv" });
      const url  = URL.createObjectURL(blob);
      const a    = document.createElement("a");
      a.href     = url;
      a.download = "trades.csv";
      a.click();
      URL.revokeObjectURL(url);
    }

    export default function TradeHistory({ trades }: Props) {
      if (trades.length === 0) {
        return (
          <div className="flex flex-col items-center justify-center gap-2 py-8 text-text-dim">
            <span className="text-2xl opacity-20">◈</span>
            <span className="text-[10px] font-mono uppercase tracking-widest">No trades yet</span>
            <span className="text-[10px] font-mono text-text-dim/50">
              Use the trade bar or ask Finance Ally to execute your first trade
            </span>
          </div>
        );
      }

      return (
        <div className="flex flex-col h-full">
          <div className="flex items-center justify-between px-3 py-1.5 border-b border-border-subtle shrink-0">
            <span className="text-[9px] font-mono text-text-dim uppercase tracking-widest">
              {trades.length} trade{trades.length !== 1 ? "s" : ""}
            </span>
            <button
              onClick={() => exportToCsv(trades)}
              className="flex items-center gap-1.5 px-2 py-0.5 text-[9px] font-mono uppercase tracking-widest text-text-dim border border-border rounded hover:border-accent hover:text-accent transition-colors"
            >
              <span>↓</span>
              <span>Export CSV</span>
            </button>
          </div>
          <div className="overflow-x-auto flex-1 min-h-0">
            <table className="w-full text-left border-collapse min-w-[540px]">
              <thead>
                <tr className="border-b border-border sticky top-0 bg-surface z-10">
                  {["Date", "Ticker", "Side", "Qty", "Price", "Total"].map((h) => (
                    <th key={h} className={`${LABEL} px-3 py-2 font-normal`}>{h}</th>
                  ))}
                </tr>
              </thead>
              <tbody>
                {trades.map((trade) => {
                  const isBuy = trade.side === "buy";
                  const dateStr = new Date(trade.executed_at).toLocaleString("en-US", {
                    month: "short", day: "2-digit",
                    hour: "2-digit", minute: "2-digit", hour12: false,
                  });
                  return (
                    <tr key={trade.id} className="border-b border-border-subtle hover:bg-surface-2 transition-colors">
                      <td className={`${CELL} px-3 py-2 text-text-dim`}>{dateStr}</td>
                      <td className={`${CELL} px-3 py-2`}>
                        <span className="text-accent font-semibold tracking-wide">{trade.ticker}</span>
                      </td>
                      <td className={`${CELL} px-3 py-2`}>
                        <span className={`px-1.5 py-0.5 rounded text-[9px] font-mono font-semibold uppercase ${
                          isBuy
                            ? "bg-up/10 border border-up/25 text-up"
                            : "bg-down/10 border border-down/25 text-down"
                        }`}>
                          {trade.side}
                        </span>
                      </td>
                      <td className={`${CELL} px-3 py-2 text-text`}>{trade.quantity}</td>
                      <td className={`${CELL} px-3 py-2 text-text`}>${trade.price.toFixed(2)}</td>
                      <td className={`${CELL} px-3 py-2 text-text`}>${trade.total.toFixed(2)}</td>
                    </tr>
                  );
                })}

    "use client";

    import type { Portfolio, PortfolioAnalytics as AnalyticsData } from "@/lib/types";

    interface Props {
      analytics: AnalyticsData | null;
      portfolio: Portfolio | null;
    }

    function fmt(n: number, decimals = 2) {
      return n.toLocaleString("en-US", {
        minimumFractionDigits: decimals,
        maximumFractionDigits: decimals,
      });
    }

    interface CardProps {
      label: string;
      value: string;
      sub?: string;
      color?: "up" | "down" | "neutral" | "blue";
    }

    function MetricCard({ label, value, sub, color = "neutral" }: CardProps) {
      const cls = {
        up:      "bg-up/[0.07]   border-up/20   text-up",
        down:    "bg-down/[0.07] border-down/20 text-down",
        neutral: "bg-surface-2   border-border   text-text",
        blue:    "bg-blue/[0.07] border-blue/20  text-blue",
      }[color];
      return (
        <div className={`flex flex-col gap-0.5 px-4 py-3 rounded border ${cls}`}>
          <span className="text-[8px] font-mono text-text-dim uppercase tracking-widest leading-none">
            {label}
          </span>
          <span className="font-mono font-semibold text-sm tabular-nums leading-tight mt-1">
            {value}
          </span>
          {sub && (
            <span className="text-[9px] font-mono text-text-dim/70 tabular-nums">{sub}</span>
          )}
        </div>
      );
    }

    export default function PortfolioAnalytics({ analytics, portfolio }: Props) {
      if (!analytics) {
        return (
          <div className="flex flex-col items-center justify-center gap-2 py-8 text-text-dim">
            <span className="text-2xl opacity-20">◈</span>
            <span className="text-[10px] font-mono uppercase tracking-widest">Loading analytics…</span>
          </div>
        );
      }

      const returnColor   = analytics.total_return_pct >= 0 ? "up" : "down";
      const realizedColor = analytics.realized_pnl     >= 0 ? "up" : "down";
      const unrealColor   = analytics.unrealized_pnl   >= 0 ? "up" : "down";
      const returnSign    = analytics.total_return_pct >= 0 ? "+" : "";
      const realizedSign  = analytics.realized_pnl     >= 0 ? "+" : "";
      const unrealSign    = analytics.unrealized_pnl   >= 0 ? "+" : "";

      return (
        <div className="flex flex-col h-full overflow-auto">
          <div className="flex items-center gap-3 px-4 py-1.5 border-b border-border-subtle shrink-0">
            <span className="text-[9px] font-mono text-text-dim tabular-nums">
              {analytics.total_trades} trades
            </span>
            <span className="text-border/40 text-[9px]">·</span>
            <span className="text-[9px] font-mono text-up/80 tabular-nums">
              {analytics.buy_count} buys
            </span>
            <span className="text-border/40 text-[9px]">·</span>
            <span className="text-[9px] font-mono text-down/80 tabular-nums">
              {analytics.sell_count} sells
            </span>
          </div>
          <div className="grid grid-cols-3 gap-2 p-4">
            <MetricCard
              label="Total Return"
              value={`${returnSign}${fmt(analytics.total_return_pct)}%`}
              sub={portfolio ? `$${fmt(portfolio.total_value)}` : undefined}
              color={returnColor}
            />
            <MetricCard
              label="Realized P&L"
              value={`${realizedSign}$${fmt(Math.abs(analytics.realized_pnl))}`}
              color={realizedColor}
            />
            <MetricCard
              label="Unrealized P&L"
              value={`${unrealSign}$${fmt(Math.abs(analytics.unrealized_pnl))}`}
              color={unrealColor}
            />
            <MetricCard
              label="Win Rate"
              value={analytics.sell_count > 0 ? `${fmt(analytics.win_rate, 1)}%` : "—"}
              sub={analytics.sell_count > 0 ? `${analytics.sell_count} closed trades` : "No sells yet"}
              color="blue"
            />
            <MetricCard
              label="Best Performer"
              value={analytics.best_performer ? analytics.best_performer.ticker : "—"}
              sub={analytics.best_performer ? `+${fmt(analytics.best_performer.pnl_percent)}%` : undefined}
              color={analytics.best_performer ? "up" : "neutral"}
            />
            <MetricCard
              label="Worst Performer"
              value={analytics.worst_performer ? analytics.worst_performer.ticker : "—"}
              sub={analytics.worst_performer ? `${fmt(analytics.worst_performer.pnl_percent)}%` : undefined}
              color={
                analytics.worst_performer && analytics.worst_performer.pnl_percent < 0
                  ? "down"
                  : "neutral"
              }
            />
          </div>
        </div>
      );
    }
  </action>
